import express from "express";
import OpenAI from "openai";

const app = express();
app.use(express.json({ limit: "1mb" }));

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

const BOT_SHARED_SECRET = process.env.BOT_SHARED_SECRET || "change-me";

app.get("/", (req, res) => {
  res.send("Tarot bridge server is running.");
});

app.post("/tarot-chat", async (req, res) => {
  try {
    const { secret, question, nickname, room } = req.body ?? {};

    if (secret !== BOT_SHARED_SECRET) {
      return res.status(403).json({ error: "forbidden" });
    }

    if (!question || typeof question !== "string") {
      return res.status(400).json({ error: "question is required" });
    }

    const cleanQuestion = question.trim();

    const systemPrompt = `
너는 전문 타로상담사다.
항상 한국어 존댓말로 답한다.
규칙:
- 질문에 바로 답한다.
- 가능성 중심으로 말한다.
- 너무 단정적으로 말하지 않는다.
- 불안만 키우는 과장된 표현은 피한다.
- 답변은 4~7문장 정도로 한다.
- 타로 상담처럼 현재 흐름, 상대 심리, 조언 느낌을 포함한다.
`;

    const userPrompt = `
질문자 닉네임: ${nickname || "익명"}
방 이름: ${room || "알 수 없음"}
질문: ${cleanQuestion}

위 질문에 대해 타로 상담처럼 자연스럽게 답변해 주세요.
`;

    const response = await client.responses.create({
      model: "gpt-5.4",
      input: [
        { role: "system", content: systemPrompt },
        { role: "user", content: userPrompt }
      ]
    });

    return res.json({
      reply: (response.output_text || "").trim()
    });
  } catch (error) {
    console.error(error);
    return res.status(500).json({
      error: "server_error",
      detail: String(error?.message || error)
    });
  }
});

const port = process.env.PORT || 3000;
app.listen(port, () => {
  console.log(`Tarot bridge server listening on ${port}`);
});
